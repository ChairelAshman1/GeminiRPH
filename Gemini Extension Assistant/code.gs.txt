/**
 * Web App API helpers for website integration
 */
function doGet(e) {
  var action = (e.parameter.action || 'ping').toLowerCase();
  switch(action) {
    case 'ping':
      return ContentService
          .createTextOutput(JSON.stringify({ status: 'ok', message: 'alive' }))
          .setMimeType(ContentService.MimeType.JSON);

    case 'read':
      return handleReadRequest(e);

    case 'listSheets':
      return handleListSheets(e);

    case 'getApiKeys':
      return handleGetApiKeys(e);

    case 'getSchedule':
      return handleGetSchedule(e);

    default:
      return ContentService
          .createTextOutput(JSON.stringify({ status: 'error', message: 'Unsupported action' }))
          .setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  var data;
  try {
    data = JSON.parse(e.postData.contents);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'Invalid JSON body: ' + err } ))
      .setMimeType(ContentService.MimeType.JSON);
  }

  var action = (data.action || '').toLowerCase();
  if (action === 'write') {
    return handleWriteRequest(data);
  }
  if (action === 'appendRPH') {
    return handleAppendRPH(data);
  }
  if (action === 'saveApiKey') {
    return handleSaveApiKey(data);
  }

  return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'Unsupported POST action' }))
      .setMimeType(ContentService.MimeType.JSON);
}

function handleReadRequest(e) {
  var sheetId = e.parameter.sheetId || '';
  var sheetName = e.parameter.sheetName || 'Sheet1';
  var range = e.parameter.range || 'A1:Z50';

  var sheet = getSheetInstance(sheetId, sheetName);
  if (!sheet) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'Sheet tidak ditemui.' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  var values = sheet.getRange(range).getValues();
  return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', values: values }))
      .setMimeType(ContentService.MimeType.JSON);
}

function handleGetApiKeys(e) {
  var sheetId = e.parameter.sheetId || '';
  var sheet = getSheetInstance(sheetId, 'API_KEYS');
  if (!sheet) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', keys: [] }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  try {
    var values = sheet.getDataRange().getValues();
    var keys = [];
    for (var i = 1; i < values.length; i++) {
      var row = values[i];
      if (row[0]) keys.push({ key: row[0], created: row[1] || '' });
    }
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', keys: keys }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function handleSaveApiKey(data) {
  var sheetId = data.sheetId || '';
  var key = (data.key || '').trim();
  if (!key) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'API Key kosong.' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  var ss = sheetId && sheetId.length > 10 ? SpreadsheetApp.openById(sheetId) : SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('API_KEYS');
  if (!sheet) {
    sheet = ss.insertSheet('API_KEYS');
    sheet.getRange('A1:B1').setValues([['key','created']]);
  }

  // Avoid duplicate
  var dataRange = sheet.getDataRange().getValues();
  for (var i = 1; i < dataRange.length; i++) {
    if (dataRange[i][0] === key) {
      return ContentService
        .createTextOutput(JSON.stringify({ status: 'ok', message: 'Key sudah wujud.' }))
        .setMimeType(ContentService.MimeType.JSON);
    }
  }

  sheet.appendRow([key, new Date().toISOString()]);
  return ContentService
    .createTextOutput(JSON.stringify({ status: 'ok', message: 'Key disimpan.' }))
    .setMimeType(ContentService.MimeType.JSON);
}

function handleGetSchedule(e) {
  var sheetId = e.parameter.sheetId || '';
  var sheetName = e.parameter.sheetName || 'Jadual';
  var sheet = getSheetInstance(sheetId, sheetName);
  if (!sheet) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'Sheet jadual tidak ditemui.' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  try {
    var data = sheet.getDataRange().getValues();
    if (data.length < 2) {
      return ContentService
        .createTextOutput(JSON.stringify({ status: 'ok', schedule: [] }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    var headers = data[0].map(function(h) { return (h || '').toString().toLowerCase(); });
    var dayIndex = headers.indexOf('day');
    var classIndex = headers.indexOf('class');
    var temaIndex = headers.indexOf('tema');
    var tajukIndex = headers.indexOf('tajuk');
    var skIndex = headers.indexOf('sk');
    var spIndex = headers.indexOf('sp');

    var schedule = [];
    for (var i = 1; i < data.length; i++) {
      var row = data[i];
      if (!row[dayIndex]) continue;
      schedule.push({
        day: row[dayIndex],
        className: row[classIndex] || '',
        tema: row[temaIndex] || '',
        tajuk: row[tajukIndex] || '',
        sk: row[skIndex] || '',
        sp: row[spIndex] || ''
      });
    }

    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', schedule: schedule }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function handleListSheets(e) {
  var sheetId = e.parameter.sheetId || '';
  if (!sheetId) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'Sheet ID diperlukan.' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  try {
    var ss = sheetId && sheetId.length > 10 ? SpreadsheetApp.openById(sheetId) : SpreadsheetApp.getActiveSpreadsheet();
    var sheetNames = ss.getSheets().map(function(s) { return s.getName(); });
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', sheets: sheetNames }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function handleAppendRPH(data) {
  var sheetId = data.sheetId || '';
  var sheetName = data.sheetName || 'RPH_OUTPUT';
  var rows = data.rows || [];

  if (!sheetId) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'Sheet ID diperlukan.' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  if (!Array.isArray(rows) || rows.length === 0) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'Tiada data untuk ditambah.' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  try {
    var ss = sheetId && sheetId.length > 10 ? SpreadsheetApp.openById(sheetId) : SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(sheetName);
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      sheet.getRange('A1:D1').setValues([['Class', 'Objektif', 'Kriteria Kejayaan', 'Aktiviti']]);
    }

    rows.forEach(function(row) {
      sheet.appendRow([row.className || '', row.objektif || '', row.kriteria || '', row.aktiviti || '']);
    });

    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', message: 'RPH disimpan dalam sheet ' + sheetName }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function handleWriteRequest(data) {
  var sheetId = data.sheetId || '';
  var sheetName = data.sheetName || 'Sheet1';
  var range = data.range || 'A2';
  var payload = data.data || {};

  var sheet = getSheetInstance(sheetId, sheetName);
  if (!sheet) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'Sheet tidak ditemui.' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  var values = [
    payload.timestamp || '',
    payload.objektif || '',
    payload.kriteria || '',
    payload.aktiviti || ''
  ];

  try {
    var target = sheet.getRange(range).offset(0, 0, 1, values.length);
    target.setValues([values]);
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'ok', message: 'Data ditulis.' }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ status: 'error', message: 'Gagal menulis: ' + err }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function getSheetInstance(sheetId, sheetName) {
  var ss;
  try {
    if (sheetId && sheetId.length > 10) {
      ss = SpreadsheetApp.openById(sheetId);
    } else {
      ss = SpreadsheetApp.getActiveSpreadsheet();
    }
    return ss.getSheetByName(sheetName) || ss.getSheets()[0];
  } catch (e) {
    return null;
  }
}
